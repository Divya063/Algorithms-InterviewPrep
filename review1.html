Problem

    Determine if a given string is a palindrome. A palindrome is a string that is the same forwards as backwards, ignoring letter case, punctuation marks and word dividers. For example, “racecar” and “Amor, Roma” are palindromes, “abcd” is not. We limit the punctuation marks to ignore by the period, comma, question mark, exclamation point, apostrophe, colon, semicolon, hyphen, white space.
    Input: String
    Output: Return True if the string is a palindrome, return False otherwise.
    Sample Input:'civic'
    Sample Output: True

Solution (C++)

    
using namespace std;

int main() {
    /* Enter your code here. Read input from STDIN. Print output to STDOUT */
    std::string input;
    char l[] = {'.', ',', '?', '!', '\'', ':', ';','-', ' '};
    
    set<char> ignoreList(l, l+sizeof(l));
    
    while (std::cin) {
        getline(std::cin, input);
    }
    locale loc;
    std::string::iterator it = input.begin();
    std::string::reverse_iterator rit = input.rbegin();
    while (it != input.end() && rit != input.rend()) {
        if (ignoreList.find(*it) != ignoreList.end()) {
            ++it;
            continue;
        }
        if (ignoreList.find(*rit) != ignoreList.end()) {
            ++rit;
            continue;
        }
        if (tolower(*it, loc) != tolower(*rit, loc)) {
            std::cout << "False" << std::endl;
            return 0;
        }
        ++it;
        ++rit;
    }
    std::cout << "True" << std::endl;
    return 0;
}

Problem

    Given 2 arrays find an intersection of them. Output array should be sorted without duplicates. Build-in function for sorting is allowed to use. The effective solution would have a time complexity O(m + n) before sorting the intersection array where n and m are sizes of the arrays. 
     
Input
size of array 1, size of array 2
array 1
array 2

Output
Print out of the sorted intersection array with whitespace as divider

Sample Input
3 4
1 2 3
0 2 3 4

Sample Output
2 3

Solution 1 (C++)

    
using namespace std;
int main() {
    /* Enter your code here. Read input from STDIN. Print output to STDOUT */   
    std::vector<int> array1;
    std::vector<int> array2;
    std::vector<int> intersect; 

    std::string line;
    int count = 0;
    
    while (getline(std::cin, line)) {
        int num;
        std::stringstream lineStr(line);
        while (lineStr >> num) {
            if (count == 1) {
                array1.push_back(num);
            } else if (count == 2) {
                array2.push_back(num);
            }
        }
        count++;
    }
    std::sort (array1.begin(), array1.end());
    std::sort (array2.begin(), array2.end());

    std::vector<int>::const_iterator it1 = array1.begin();
    std::vector<int>::const_iterator it2 = array2.begin();

    while (it1 != array1.end() && it2 != array2.end()) {
        if (*it1 < *it2) {
            ++it1;
        } else if (*it2 < *it1) {
            ++it2;
        } else {
            if (intersect.empty() || intersect.back() != *it1) {
                intersect.push_back(*it1);
            }
            ++it1;
            ++it2;
        }
    }
    
    for (int i=0; i < intersect.size(); i++) {
        std::cout << intersect[i] << ' ';
    }
    
    return 0;
}

Solution 2 (C++)

    
using namespace std;
int main() {
    /* Enter your code here. Read input from STDIN. Print output to STDOUT */
    std::string input;
    std::vector<int> array1;
    std::vector<int> array2;
    int count = 0;
    
    while (getline (std::cin, input)) {
        std::stringstream stream(input);
        int num;
        while (stream >> num) {
            if (count == 1) {
                array1.push_back(num);
            } else if (count == 2) {
                array2.push_back(num);
            }
        }
        count++;
    }

    std::map<int, int> table;
    std::vector<int> intersection;
    
    // insert in the table
    for (int i=0; i < array1.size(); i++) {
        if (table[array1[i]] == 0) {
            table[array1[i]]++;    
        }
    }
    
    // second pass to check duplicates
    for (int i=0; i < array2.size(); i++) {
        if (table[array2[i]] == 1) {
            intersection.push_back(array2[i]);
            table[array2[i]]++;
        }
    }

    sort(intersection.begin(), intersection.end());
    for (int i=0; i < intersection.size(); i++) {
        std::cout << intersection[i] << ' ';
    }
    
    return 0;
}
